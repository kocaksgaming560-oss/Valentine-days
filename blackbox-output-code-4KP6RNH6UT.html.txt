    <script>
        let sections = document.querySelectorAll('.section');
        let surprise = document.getElementById('surprise');
        let scrollCount = 0;

        function startJourney() {
            document.querySelector('.intro').style.display = 'none';
            sections.forEach((section, index) => {
                setTimeout(() => {
                    section.classList.add('show');
                }, index * 1000);
            });
        }

        window.addEventListener('scroll', () => {
            scrollCount++;
            if (scrollCount > 10) { // Simulasi scroll hingga akhir
                surprise.classList.add('show');
            }
        });

        // Hover effects
        document.querySelectorAll('.memory').forEach(memory => {
            memory.addEventListener('mouseenter', () => {
                memory.style.transform = 'scale(1.05)';
            });
            memory.addEventListener('mouseleave', () => {
                memory.style.transform = 'scale(1)';
            });
        });
    </script>
